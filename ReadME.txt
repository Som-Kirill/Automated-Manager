    config = {
        'token': '-',

            !Префікс команд те що ставится перед командами
        'channel_sending_data_time_id': 1136635635172638803,

            !ID чату для надсилання запиту на здійснення деяких команд

            !ID АФК чату якщо не буде вказано то бот буде рахувати кількість проведеного чалу в АФК чаті
        'channel_is_another_game_id': 1138913952844226630,
            !ID категорій інша гра в якій будуть створюватись канали
        'the_number_of_minutes_required_to_save': 1,
            !Мінімальна кількість секунд необхідна для збереження даних

            !Сортуванняв команді alltime на категорії добре нормально погано в зміній треба вказувати мінімальну кількість секунд необхідну 
            для збереження в відповідній катигорії

        'role_clear_id': 1138140321344139326,
